步骤 0: 初始状态 - 准备交换
// 初始值定义（C++）
int a = 10; // 变量a初始化为10
int b = 25; // 变量b初始化为25
int temp; // 声明临时变量temp（未初始化）
 
// 交换过程（C++经典三步法）
temp = a; // 步骤1: 将a的值保存到temp
a = b; // 步骤2: 将b的值赋给a
b = temp; // 步骤3: 将temp的值赋给b
